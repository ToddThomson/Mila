107void manual_seed(mt19937_state* state, unsigned int seed) {
108 state->MATRIX_A[0] = 0x0u;
109 state->MATRIX_A[1] = 0x9908b0df;
110 state->state_[0] = seed & 0xffffffff;
111 for (unsigned int j = 1; j < MERSENNE_STATE_N; j++) {
112 state->state_[j] = 1812433253 * (state->state_[j - 1] ^ (state->state_[j - 1] >> 30)) + j;
113 state->state_[j] &= 0xffffffff;
114 }
115 state->left_ = 1;
116 state->next_ = 0;
117}

119void next_state(mt19937_state* state) {
120 state->left_ = MERSENNE_STATE_N;
121 state->next_ = 0;
122 unsigned int y, j;
123 for (j = 0; j < MERSENNE_STATE_N - MERSENNE_STATE_M; j++) {
124 y = (state->state_[j] & UMASK) | (state->state_[j + 1] & LMASK);
125 state->state_[j] = state->state_[j + MERSENNE_STATE_M] ^ (y >> 1) ^ state->MATRIX_A[y & 0x1];
126 }
127 for (; j < MERSENNE_STATE_N - 1; j++) {
128 y = (state->state_[j] & UMASK) | (state->state_[j + 1] & LMASK);
129 state->state_[j] = state->state_[j + (MERSENNE_STATE_M - MERSENNE_STATE_N)] ^ (y >> 1) ^ state->MATRIX_A[y & 0x1];
130 }
131 y = (state->state_[MERSENNE_STATE_N - 1] & UMASK) | (state->state_[0] & LMASK);
132 state->state_[MERSENNE_STATE_N - 1] = state->state_[MERSENNE_STATE_M - 1] ^ (y >> 1) ^ state->MATRIX_A[y & 0x1];
133}

135unsigned int randint32(mt19937_state* state) {
136 if (!state) return 0;
137 if (state->MATRIX_A[0] != 0 || state->MATRIX_A[1] != 0x9908b0df) manual_seed(state, 5489); // auto-initialize
138 if (--state->left_ <= 0) {
139 next_state(state);
140 }
141 unsigned int y = state->state_[state->next_++];
142 y ^= y >> 11;
143 y ^= (y << 7) & 0x9d2c5680;
144 y ^= (y << 15) & 0xefc60000;
145 y ^= y >> 18;
146 return y;
147}

167// Box-Muller transform: maps uniform random numbers to Gaussian distributed numbers
168// https://en.wikipedia.org/wiki/Box%E2%80%93Muller_transform
169void normal_fill_16(float* data, float mean, float std) {
170 #define EPSILONE 1e-12f
171 for (unsigned int t = 0; t < 8; t++) {
172 float u1 = 1 - data[t];
173 float u2 = data[t + 8];
174 float radius = sqrtf(-2 * logf(u1 + EPSILONE));
175 float theta = (float) (2.0 * M_PI * u2);
176 data[t] = (radius * cosf(theta) * std + mean);
177 data[t + 8] = (radius * sinf(theta) * std + mean);
178 }
179}

181void normal_fill(float* data, unsigned int numel, float mean, float std, mt19937_state* state) {
182 for (unsigned int t = 0; t < numel; t++) {
183 data[t] = randfloat32(state);
184 }
185 for (unsigned int i = 0; i < numel - 15; i += 16) {
186 normal_fill_16(data + i, mean, std);
187 }
188 if (numel % 16 != 0) {
189 // recompute the last 16 values
190 data = data + numel - 16;
191 for (unsigned int i = 0; i < 16; i++) {
192 data[i] = randfloat32(state);
193 }
194 normal_fill_16(data, mean, std);
195 }
196}

198void normal_(float* data, unsigned int numel, float mean, float std, mt19937_state* state) {
199 #define EPSILONE 1e-12f
200 if (numel >= 16) {
201 normal_fill(data, numel, mean, std, state);
202 }
203 else {
204 double next_double_normal_sample = 0.0; // make compiler warning happy, won't be used
205 int has_next_double_normal_sample = 0;
206 for (unsigned int t = 0; t < numel; t++) {
207 if (has_next_double_normal_sample) {
208 data[t] = (float)(next_double_normal_sample * std + mean);
209 has_next_double_normal_sample = 0;
210 continue;
211 }
212 // for numel < 16 we draw a double (float64)
213 float u1 = (float) randfloat64(state);
214 float u2 = (float) randfloat64(state);
215 float radius = sqrtf(-2 * logf(1 - u2 + EPSILONE));
216 float theta = (float) (2.0 * M_PI * u1);
217 next_double_normal_sample = radius * sinf(theta);
218 has_next_double_normal_sample = 1;
219 data[t] = (radius * cosf(theta) * std + mean);
220 }
221 }

230void random_permutation(int* data, int numel, mt19937_state* state) {
231 for (int i = numel - 1; i > 0; i--) {
232 // pick an index j in [0, i] with equal probability
233 int j = randint32(state) % (i + 1);
234 // swap i <-> j
235 int tmp = data[i];
236 data[i] = data[j];
237 data[j] = tmp;
238 }
239}
